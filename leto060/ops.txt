BEGIN	ori_ccr
	fetch_l(next);
	ccr.set(ccr.get() | next);
END

BEGIN	ori_b_reg
	fetch_b(next);
	D(rrr) = (D(rrr) & 0xffffff00) | (next & 0xff);
	ccr.setMOVEB(D(rrr));
END

BEGIN	ori_w_reg
	fetch_w(next)
	D(rrr) = (D(rrr) & 0xffff0000) | next;
	ccr.setMOVEW(D(rrr));
END

BEGIN	ori_l_reg
	fetch_l(next)
	D(rrr) = D(rrr) | next;
	ccr.setMOVEW(D[rrr]);
END

BEGIN	ori_b_mem
	fetch_b(next);
	EA(addr);
	read_b(addr, data);
	data |= (next & 0xff);
	ccr.setMOVEB(data);
	write_b(addr, data);
END
